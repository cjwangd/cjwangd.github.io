testret(){

   if [ $1 -ne 0 ]; then
      exit 1;
   fi
   printf  "\033[36m \t\t\t [OK] \033[0m\n" 

}

unsetenv(){

   unset CORE_PEER_LOCALMSPID
   unset CORE_PEER_ADDRESS
   unset CORE_PEER_MSPCONFIGPATH

PEER_PIDS=$(ps -ef|grep 'peer node start'|grep -v grep|awk '{print $2}')
